Complete Form Payload Structure:
interface BrandFormPayload {
  // Basic Information (Required)
  name: string;                    // âœ… Required
  description: string;             // âœ… Required
  shortDescription?: string;       // Optional
  
  // Visual Assets (Optional)
  logo?: File | string;            // Image upload or URL
  banner?: File | string;          // Image upload or URL
  
  // Business Information (Optional)
  businessInfo?: {
    foundingYear?: number;         // 1800 - current year
    originCountry?: string;        // Country select
    headquarters?: string;         // Text input
    parentCompany?: string;        // Text input
    legalName?: string;            // Text input
    registrationNumber?: string;   // Text input
    taxId?: string;                // Text input
  };
  
  // Social Media (Optional)
  socialMedia?: {
    website?: string;              // URL validation
    facebook?: string;             // Facebook URL validation
    instagram?: string;            // Instagram URL validation
    twitter?: string;              // Twitter URL validation
    linkedin?: string;             // LinkedIn URL validation
    youtube?: string;              // YouTube URL validation
    tiktok?: string;               // TikTok URL validation
  };
  
  // SEO (Optional)
  seo?: {
    metaTitle?: string;            // Max 60 chars
    metaDescription?: string;      // Max 160 chars
    metaKeywords?: string[];       // Tag input, max 20
  };
  
  // Display Settings (Optional)
  order?: number;                  // 0-9999
  isActive?: boolean;              // Default: true
  isFeatured?: boolean;            // Default: false
  showInMenu?: boolean;            // Default: true
  showInHomepage?: boolean;        // Default: false
}


Form Validation Rules:
const BrandFormValidation = {
  // Required Fields
  name: {
    required: true,
    minLength: 2,
    maxLength: 100,
    pattern: /^[a-zA-Z0-9\s'&.-]+$/, // Allow common brand name characters
  },
  description: {
    required: true,
    maxLength: 2000,
  },
  
  // Optional Field Limits
  shortDescription: { maxLength: 300 },
  'businessInfo.foundingYear': { 
    min: 1800, 
    max: new Date().getFullYear() 
  },
  'businessInfo.originCountry': { maxLength: 100 },
  'businessInfo.headquarters': { maxLength: 200 },
  'businessInfo.parentCompany': { maxLength: 100 },
  'businessInfo.legalName': { maxLength: 200 },
  'businessInfo.registrationNumber': { maxLength: 50 },
  'businessInfo.taxId': { maxLength: 50 },
  
  // SEO Limits
  'seo.metaTitle': { maxLength: 60 },
  'seo.metaDescription': { maxLength: 160 },
  'seo.metaKeywords': { maxItems: 20 },
  
  // Display Settings
  order: { min: 0, max: 9999 },
  
  // URL Validations
  'socialMedia.website': { pattern: /^https?:\/\/.+/ },
  'socialMedia.facebook': { pattern: /^https?:\/\/(www\.)?facebook\.com\/.+/ },
  'socialMedia.instagram': { pattern: /^https?:\/\/(www\.)?instagram\.com\/.+/ },
  'socialMedia.twitter': { pattern: /^https?:\/\/(www\.)?(twitter\.com|x\.com)\/.+/ },
  'socialMedia.linkedin': { pattern: /^https?:\/\/(www\.)?linkedin\.com\/.+/ },
  'socialMedia.youtube': { pattern: /^https?:\/\/(www\.)?youtube\.com\/.+/ },
  'socialMedia.tiktok': { pattern: /^https?:\/\/(www\.)?tiktok\.com\/.+/ },
};
